65  public IList<IAsset> Filter(IList<IAsset> Assets)
66  {
67  if (Assets == null || Assets.Count == 0)
68  return new List<IAsset>();
69  if (Assets.FirstOrDefault().Type != AssetType.CSS)
70  return Assets;
71  FileManager FileSystem = BatComputer.Bootstrapper.Resolve<FileManager>();
72  foreach (IAsset Asset in Assets)
73  {
74  foreach (Match ImageMatch in ImageRegex.Matches(Asset.Content))
75  {
76  string TempFile = ImageMatch.Groups["File"].Value;
77  string MatchString = ImageMatch.Value;
78  IFile File = FileSystem.File(TempFile);
79  File = DetermineFile(File, FileSystem, Asset, TempFile);
80  if (File == null || !File.Exists)
81  {
82  IFile AssetFile = FileSystem.File(Asset.Path);
83  File = FileSystem.File(AssetFile.Directory.FullName + "\\" + TempFile);
84  }
85  if (File.Exists)
86  {
87  if (File.Extension.ToUpperInvariant() == ".TTF"
88  || File.Extension.ToUpperInvariant() == ".OTF"
89  || File.Extension.ToUpperInvariant() == ".WOFF"
90  || File.Extension.ToUpperInvariant() == ".SVG"
91  || File.Extension.ToUpperInvariant() == ".EOT")
92  {
93  string MIME = "";
94  if (File.Extension.ToUpperInvariant() == ".WOFF")
95  MIME = "application/x-font-woff";
96  else if (File.Extension.ToUpperInvariant() == ".OTF")
97  MIME = "application/x-font-opentype";
98  else if (File.Extension.ToUpperInvariant() == ".TTF")
99  MIME = "application/x-font-ttf";
100  else if (File.Extension.ToUpperInvariant() == ".SVG")
101  MIME = "image/svg+xml";
102  else if (File.Extension.ToUpperInvariant() == ".EOT")
103  MIME = "application/vnd.ms-fontobject";
104 
105  Asset.Content = Asset.Content.Replace(MatchString, "url(data:" + MIME + ";base64," + File.ReadBinary().ToString(Base64FormattingOptions.None) + ")");
106  }
107  else
108  {
109  try
110  {
111  using (Bitmap TempImage = new Bitmap(File.FullName))
112  {
113  string MIMEType = "image/jpeg";
114  string Content = "";
115  if (File.FullName.ToUpperInvariant().EndsWith(".PNG"))
116  {
117  MIMEType = "image/png";
118  Content = TempImage.ToBase64(ImageFormat.Png);
119  }
120  else if (File.FullName.ToUpperInvariant().EndsWith(".JPG") || File.FullName.ToUpperInvariant().EndsWith(".JPEG"))
121  {
122  MIMEType = "image/jpeg";
123  Content = TempImage.ToBase64(ImageFormat.Jpeg);
124  }
125  else if (File.FullName.ToUpperInvariant().EndsWith(".GIF"))
126  {
127  MIMEType = "image/gif";
128  Content = TempImage.ToBase64(ImageFormat.Gif);
129  }
130  else if (File.FullName.ToUpperInvariant().EndsWith(".TIFF"))
131  {
132  MIMEType = "image/tiff";
133  Content = TempImage.ToBase64(ImageFormat.Tiff);
134  }
135  else if (File.FullName.ToUpperInvariant().EndsWith(".BMP"))
136  {
137  MIMEType = "image/bmp";
138  Content = TempImage.ToBase64(ImageFormat.Bmp);
139  }
140  Asset.Content = Asset.Content.Replace(MatchString, "url(data:" + MIMEType + ";base64," + Content + ")");
141  }
142  }
143  catch { }
144  }
145  }
146  }
147  }
148  return Assets;
149  }
