125  public AssetType DetermineType(string Path)
126  {
127  AssetType Type = Translators.FirstOrDefault(x => Path.EndsWith(x.FileTypeAccepts))
128  .Chain(x => x.TranslatesTo, AssetType.Unknown);
129  if (Type == AssetType.Unknown && Path.EndsWith("css"))
130  return AssetType.CSS;
131  else if (Type == AssetType.Unknown && Path.EndsWith("js"))
132  return AssetType.Javascript;
133  return Type;

142  public void Process(IList<IAsset> Assets, BundleContext Context, BundleResponse Response)
143  {
144  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == RunTime.PreTranslate))
145  {
146  Assets = Filter.Filter(Assets);
147  }
148  foreach (ITranslator Translator in Translators)
149  {
150  Assets = Translator.Translate(Assets);
151  }
152  foreach (RunTime Item in RunOrder)
153  {
154  foreach (IFilter Filter in Filters.Where(x => x.TimeToRun == Item))
155  {
156  Assets = Filter.Filter(Assets);
157  }
158  }
159  string Content = Assets.OrderBy(x => x.Path).ToString(x => x.ToString(), "");
160  foreach (IContentFilter Filter in ContentFilters)
161  {
162  Content = Filter.Filter(Content);
163  }
164  System.Collections.Generic.List<BundleFile> Files = new System.Collections.Generic.List<BundleFile>();
165  foreach (IAsset Asset in Assets)
166  {
167  if (Asset.Path.StartsWith("~") || Asset.Path.StartsWith("/"))
168  {
169  Files.Add(new BundleFile(Asset.Path, new VirtualFileHack(Asset.Path)));
170  Files.Add(Asset.Included.Where(x => x.Path.StartsWith("~") || x.Path.StartsWith("/")).Select(x => new BundleFile(x.Path, new VirtualFileHack(x.Path))));
171  }
172  }
173  Response.Content = Content;
174  Response.Files = Files.OrderBy(x => x.VirtualFile.VirtualPath);
175  Response.ContentType = Assets.First().Type == AssetType.CSS ? "text/css" : "text/javascript";
176  Response.Cacheability = HttpCacheability.ServerAndPrivate;
177  }

196  private void CreateBundles(IDirectory Directory)
197  {
198  if (!Directory.Exists)
199  return;
200  string BundleDirectory = Directory.FullName.Replace(FileManager.Directory("~/").FullName, "~/").Replace("\\", "/");
201  StyleBundle Bundle = new StyleBundle(BundleDirectory + "/bundle/css");
202  Bundle.Transforms.Clear();
203  Bundle.Transforms.Add(new Transformer());
204  if (Directory.Exists)
205  {
206  foreach (string Value in FileTypes[AssetType.CSS])
207  {
208  Bundle.IncludeDirectory(BundleDirectory, "*." + Value, true);
209  }
210  }
211  ScriptBundle Bundle2 = new ScriptBundle(BundleDirectory + "/bundle/js");
212  Bundle2.Transforms.Clear();
213  Bundle2.Transforms.Add(new Transformer());
214  if (Directory.Exists)
215  {
216  foreach (string Value in FileTypes[AssetType.Javascript])
217  {
218  Bundle2.IncludeDirectory(BundleDirectory, "*." + Value, true);
219  }
220  }
221  BundleTable.Bundles.Add(Bundle);
222  BundleTable.Bundles.Add(Bundle2);
223  foreach (IDirectory SubDirectory in Directory.EnumerateDirectories("*", SearchOption.TopDirectoryOnly))
224  {
225  CreateBundles(SubDirectory);
226  }
