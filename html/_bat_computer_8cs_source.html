62  public static void PreStart()
63  {
64  new DirectoryInfo(AppDomain.CurrentDomain.BaseDirectory).LoadAssemblies(false).ForEach(x => x);
65  Bootstrapper = AppDomain.CurrentDomain.GetAssemblies().Objects<IBootstrapper>().FirstOrDefault();
66  Bootstrapper.Register<IBootstrapper>(Bootstrapper);
67  AppDomain.CurrentDomain.GetAssemblies().Objects<IModule>().ForEach(x => x.Load(Bootstrapper));
68  LogBase Logger = Bootstrapper.Resolve<LogBase>(new NullLogger());
69  Logger.LogMessage("Batman starting", MessageType.Info);
