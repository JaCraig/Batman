52  public override void OnActionExecuting(ActionExecutingContext filterContext)
53  {
54  HttpContextBase Context = filterContext.HttpContext;
55  Context.Response.Cache.VaryByHeaders["Accept-Encoding"] = true;
56  DateTime Date = GetDateLastModified(Context);
57  string Etag = GetETag(Context);
58  string IncomingEtag = Context.Request.Headers["If-None-Match"];
59  string ModifiedSince = Context.Request.Headers["If-Modified-Since"];
60  DateTime ModifiedSinceDate = DateTime.Now;
61  DateTime.TryParse(ModifiedSince, out ModifiedSinceDate);
62 
63  Date = DateTime.Parse(Date.ToString("r"));
64 
65  Context.Response.Cache.SetLastModified(Date);
66  Context.Response.Cache.SetExpires(DateTime.Now.ToUniversalTime().AddDays(DaysToCache));
67  Context.Response.Cache.SetCacheability(HttpCacheability.Public);
68  Context.Response.Cache.SetMaxAge(new TimeSpan(DaysToCache, 0, 0, 0));
69  Context.Response.Cache.SetRevalidation(HttpCacheRevalidation.None);
70  Context.Response.Cache.SetETag(Etag);
71  if (String.Compare(IncomingEtag, Etag) == 0
72  || Date.CompareTo(ModifiedSinceDate) <= 0)
73  {
74  Context.Response.StatusCode = (int)HttpStatusCode.NotModified;
75  Context.ApplicationInstance.CompleteRequest();
76  return;
77  }
78  base.OnActionExecuting(filterContext);
79  }
